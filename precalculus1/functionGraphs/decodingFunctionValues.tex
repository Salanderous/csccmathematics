\begin{example}
Several values of $H(t)$ have been encoded in the graph below.  Use the graph to evaluate $H$.
\begin{image}
\begin{tikzpicture}
\begin{axis}[
            domain=0:4, ymax=5, xmax=5, ymin=-5, xmin=-5,
            axis lines =center, xlabel=$t$, ylabel=$y$,
            every axis y label/.style={at=(current axis.above origin),anchor=south},
            every axis x label/.style={at=(current axis.right of origin),anchor=west},
            axis on top,
          ]

          
        \addplot [color=penColor2,only marks,mark=*] coordinates{(-3,2)};
        \addplot [color=penColor2,only marks,mark=*] coordinates{(0,-2)};
        \addplot [color=penColor2,only marks,mark=*] coordinates{(1,-3)};
        \addplot [color=penColor2,only marks,mark=*] coordinates{(-2,2)};

\end{axis}
\end{tikzpicture}
\end{image}

\begin{itemize}
\item $H(-3)=\answer{2}$
\item $H(0)=\answer{-2}$
\item $H(1)=\answer{-3}$
\item $H(-2)=\answer{2}$
\end{itemize}



\end{example}
