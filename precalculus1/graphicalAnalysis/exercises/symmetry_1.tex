\begin{exercise}  

Which shifting values will return the graph to the graph of $y=cos(x)$?

\begin{selectAll}
\choice {$\frac{\pi}{3}$}
\choice {$\frac{\pi}{2}$}
\choice {$\frac{5\pi}{6}$}
\choice {$\pi$}
\choice {$\frac{3\pi}{2}$}
\choice {$\frac{7\pi}{4}$}
\choice [correct]{$2\pi$}
\end{selectAll}


\end{exercise}
