\begin{question}



Let 

\[
B(t) = 4 \log_{2}(5t - 3) - 1
\]



\[
\lim\limits_{t \to \infty} B(t) = \answer{\infty}
\]



\[
\lim\limits_{t \to \frac{3}{5}^+} B(t) = \answer{-1}
\]

\end{question}
