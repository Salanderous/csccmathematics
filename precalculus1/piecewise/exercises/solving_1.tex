\begin{exercise}

Evaluate the following. (Use "DNE" for Does Not Exist.)


\begin{itemize}
\item $f(-8) = \answer{DNE}$  
\item $f(-4) = \answer{-3}$ 
\item $f(-4) = \answer{0}$ 
\item $f(-3.5) = \answer{-1}$ 
\item $f(-3) = \answer{5}$ 
\item $f(-2.5) = \answer{-3}$ 
\item $f(0) = \answer{-2}$ 
\item $f(4) = \answer{0}$ 
\item $f(7) = \answer{DNE}$ 
\end{itemize}

\end{exercise}
