\begin{exercise}
  Consider the vectors below:
  \begin{image}
    \begin{tikzpicture}
        \begin{axis}[
            unit vector ratio*=1 1 1,
            axis lines=none
          ]
          \addplot[very thick,penColor,->] plot coordinates {(0,0) (0,2)};
          \addplot[very thick,penColor2,->] plot coordinates {(0,0) (4,1)};
          \node[left] at (axis cs:0, 1) [penColor] {$\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$};
          \node[above] at (axis cs:2, .5) [penColor2] {$\overset{\boldsymbol{\rightharpoonup}}{\mathbf{w}}$};
        \end{axis}
    \end{tikzpicture}
  \end{image}
  Does $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}} \boldsymbol\times \overset{\boldsymbol{\rightharpoonup}}{\mathbf{w}}$ point toward you, or away from you, or
  neither?
  \begin{multipleChoice}
    \choice{toward me}
    \choice[correct]{away from me}
    \choice{neither}
  \end{multipleChoice}
\end{exercise}
