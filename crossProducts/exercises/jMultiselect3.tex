\begin{exercise}
Select all of the following statements that \emph{must} be true.

\begin{selectAll}
\choice [correct] {If $a >0$ is a scalar, then $| a \overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} | = a |\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} |$}
\choice [correct] {Given any nonzero vector $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}$, the vector $\frac{ \overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} } {|\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}|} $ has magnitude 1.}
\choice [correct] {Suppose that $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}$ and $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$ are nonzero three dimensional vectors and $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} \cdot \overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}} =0$. Then, $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}$ and $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$ are orthogonal.}
\choice[correct]{If $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$ is a unit vector and $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}$ is any nonzero vector, $scal_{\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}} \overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} = \overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} \cdot \overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$. }
\choice[correct]{If $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}$ and $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$ are othogonal unit vectors, then $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} \times \overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$ is a unit vector.}
\end{selectAll}

\begin{hint}
Remember that $proj_{\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}} \overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}$ is in the direction of $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$.
\end{hint}

\begin{hint}
Since $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}$ and $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$ are parallel then there is some $c$ such that $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} = c \overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$.
\end{hint}

\begin{hint}
Remember that $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} \times \overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$ is orthogonal to both $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$ and $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}$.
\end{hint}
\end{exercise}
