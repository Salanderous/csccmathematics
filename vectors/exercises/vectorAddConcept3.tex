\outcome{Add and subtract vectors.}

\begin{document}
\begin{exercise}
  Consider the vectors below:
  \begin{image}
    \begin{tikzpicture}
        \begin{axis}[
            unit vector ratio*=1 1 1,
            axis lines=none
          ]
          \addplot[very thick,penColor,->] plot coordinates {(0,0) (-2,2)};
          \addplot[very thick,penColor2,->] plot coordinates {(0,0) (-1,-1)};
          \addplot[very thick,penColor3,->] plot coordinates {(0,0) (-3,1)};
          \addplot[very thick,penColor4,->] plot coordinates {(0,0) (-3,0)};
          \addplot[very thick,penColor5,->] plot coordinates {(0,0) (-3,-1)};
          \node[above right] at (axis cs:-1, 1) [penColor] {$\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$};
          \node[right] at (axis cs:-.5, -.5) [penColor2] {$\overset{\boldsymbol{\rightharpoonup}}{\mathbf{w}}$};
          \node[above right] at (axis cs:-1.5, .5) [penColor3] {$\overset{\boldsymbol{\rightharpoonup}}{\mathbf{a}}$};
          \node[above] at (axis cs:-1.5,0) [penColor4] {$\overset{\boldsymbol{\rightharpoonup}}{\mathbf{b}}$};
          \node[below right] at (axis cs:-1.5, -.5) [penColor5] {$\overset{\boldsymbol{\rightharpoonup}}{\mathbf{c}}$};
        \end{axis}
    \end{tikzpicture}
  \end{image}
  Which of the following is equal to $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}+\overset{\boldsymbol{\rightharpoonup}}{\mathbf{w}}$?
  \begin{multipleChoice}
    \choice[correct]{Vector $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{a}}$.}
    \choice{Vector $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{b}}$.}
    \choice{Vector $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{c}}$.}
    \choice{None of these.}
  \end{multipleChoice}
\end{exercise}
