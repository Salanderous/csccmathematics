\begin{exercise}
Select all of the following statements that must be true:

\begin{selectAll}
\choice[correct]{Given any nonzero vector $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}$, the vector $\frac{\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}}{|\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}|}$ has magnitude $1$.}

\choice{Given any nonzero vectors $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}$ and $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$ with the same dimension, $|\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}+\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}| = |\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}| + |\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}|$.}

\choice[correct]{If $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} = 3\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$, then  $|\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}+\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}| = |\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}| + |\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}|$}

\choice[correct]{If $|\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}| = 5$, then $|6\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}| = 30$.}

\choice[correct]{If $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} = \left\langle u_1,u_2,u_3 \right\rangle$ and $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}} = \vector{v_1,v_2,v_3}$, then $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} = \overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$ if and only if:

\[
u_1=v_1 \qquad u_2=v_2 \qquad u_3=v_3 \qquad 
\]}

\choice{If $|\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}| = |\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}|$, then $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} = \overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$.}

\choice[correct]{If $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}} = \overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$, then $|\overset{\boldsymbol{\rightharpoonup}}{\mathbf{u}}| = |\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}|$.}
\end{selectAll}

%add a hint
\end{exercise}
