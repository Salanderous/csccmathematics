      \end{axis}

  \begin{prompt}

    \overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}} + \overset{\boldsymbol{\rightharpoonup}}{\mathbf{w}} = \left\langle  \answer{1},\answer{1} \right\rangle

  \end{prompt}
