\begin{exercise}
  Consider the vectors below:
  \begin{image}
    \begin{tikzpicture}
        \begin{axis}[
            unit vector ratio*=1 1 1,
            axis lines=none
          ]
          \addplot[very thick,penColor,->] plot coordinates {(0,0) (-1,-2)};
          \addplot[very thick,penColor2,->] plot coordinates {(0,0) (2,-1)};
          \node[left] at (axis cs:-.5, -1) [penColor] {$\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$};
          \node[above] at (axis cs:1, -.5) [penColor2] {$\overset{\boldsymbol{\rightharpoonup}}{\mathbf{w}}$};
        \end{axis}
    \end{tikzpicture}
  \end{image}
  Is $\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}} \bullet \overset{\boldsymbol{\rightharpoonup}}{\mathbf{w}}$ positive, negative, or zero?
  \begin{multipleChoice}
    \choice{positive}
    \choice{negative}
    \choice[correct]{zero}
  \end{multipleChoice}
\end{exercise}
