\begin{exercise}
  Consider the vectors below:
  \begin{image}
    \begin{tikzpicture}
      \begin{axis}[
          xmin=-5,xmax=5,ymin=-5,ymax=5,
          clip=false,
          unit vector ratio*=1 1 1,
          %axis lines=none,
          grid = major,
          ytick={-5,-4,...,5},
	  xtick={-5,-4,...,5},
          ticks = none,
        ]
          \addplot[very thick,penColor,->] plot coordinates {(0,0) (-1.49,-.75)};
          \addplot[very thick,penColor2,->] plot coordinates {(0,0) (3.51,-1.96)};
          \node[above] at (axis cs:-.745, -.375) [penColor] {$\overset{\boldsymbol{\rightharpoonup}}{\mathbf{w}}$};
          \node[above] at (axis cs:1.75, -.98) [penColor2] {$\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}$};
        \end{axis}
    \end{tikzpicture}
  \end{image}
  Assuming that the components of $\mathbf{proj}_{\overset{\boldsymbol{\rightharpoonup}}{\mathbf{w}}}(\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}})$ are whole
  numbers,
  \[
  \mathbf{proj}_{\overset{\boldsymbol{\rightharpoonup}}{\mathbf{w}}}(\overset{\boldsymbol{\rightharpoonup}}{\mathbf{v}}) = \left< \answer{2},\answer{1} \right>
  \]
  You should be able to answer this question \textbf{without} any
  computation.  
\end{exercise}
